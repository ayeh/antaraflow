# MOM Wizard Flow Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Redesign the Meetings page into a unified 5-step wizard (Setup, Attendees, Inputs, Review, Finalize) that replaces create/show/edit with one cohesive view.

**Architecture:** Alpine.js + Blade wizard with semi-linear step navigation. New Project entity. Existing routes preserved but views completely replaced. Each wizard step is a Blade partial loaded via Alpine x-show tabs. AJAX auto-save for attendees, inputs, and action items.

**Tech Stack:** Laravel 12, Alpine.js 3, Blade, Tailwind CSS v4, Pest 4

**Design Doc:** `docs/plans/2026-03-02-mom-wizard-flow-design.md`

```php
<?php

namespace App\Domain\Meeting\Services;

use App\Domain\Meeting\Models\MinutesOfMeeting;

class MomNumberService
{
    public function generate(int $organizationId): string
    {
        $year = date('Y');
        $prefix = "MOM-{$year}-";

        $lastNumber = MinutesOfMeeting::withoutGlobalScopes()
            ->where('organization_id', $organizationId)
            ->where('mom_number', 'like', "{$prefix}%")
            ->orderByDesc('mom_number')
            ->value('mom_number');

        if ($lastNumber) {
            $sequence = (int) str_replace($prefix, '', $lastNumber) + 1;
        } else {
            $sequence = 1;
        }

        return $prefix . str_pad($sequence, 6, '0', STR_PAD_LEFT);
    }
}
```

```php
public function show(MinutesOfMeeting $meeting): View
{
    $this->authorize('view', $meeting);

    $meeting->load([
        'createdBy', 'project', 'tags',
        'attendees.user', 'actionItems.assignedTo',
        'inputs', 'transcriptions', 'manualNotes',
        'extractions', 'aiConversations',
    ]);

    $isEditable = in_array($meeting->status, [MeetingStatus::Draft, MeetingStatus::InProgress]);

    $orgMembers = User::where('current_organization_id', auth()->user()->current_organization_id)
        ->orderBy('name')
        ->get(['id', 'name', 'email']);

    // Stats for each step
    $attendeeStats = [
        'total' => $meeting->attendees->count(),
        'present' => $meeting->attendees->where('is_present', true)->count(),
        'absent' => $meeting->attendees->where('is_present', false)->count(),
        'confirmed' => $meeting->attendees->where('rsvp_status', RsvpStatus::Accepted)->count(),
    ];

    $actionItemStats = [
        'total' => $meeting->actionItems->count(),
        'completed' => $meeting->actionItems->where('status', ActionItemStatus::Completed)->count(),
        'in_progress' => $meeting->actionItems->where('status', ActionItemStatus::InProgress)->count(),
        'overdue' => $meeting->actionItems->filter(fn ($ai) =>
            $ai->due_date && $ai->due_date->isPast() &&
            ! in_array($ai->status, [ActionItemStatus::Completed, ActionItemStatus::Cancelled])
        )->count(),
    ];

    return view('meetings.show', compact(
        'meeting', 'isEditable', 'orgMembers',
        'attendeeStats', 'actionItemStats',
    ));
}
```

```php
// Route
Route::post('/meetings/{meeting}/attendees/import-project', [AttendeeController::class, 'importProject'])
    ->name('meetings.attendees.import-project');

// Controller method
public function importProject(MinutesOfMeeting $meeting): JsonResponse
{
    $this->authorize('update', $meeting);

    if (! $meeting->project_id) {
        return response()->json(['message' => 'No project assigned'], 422);
    }

    $projectMembers = $meeting->project->members;
    $added = 0;

    foreach ($projectMembers as $member) {
        $exists = $meeting->attendees()
            ->where('user_id', $member->id)
            ->exists();

        if (! $exists) {
            $meeting->attendees()->create([
                'user_id' => $member->id,
                'name' => $member->name,
                'email' => $member->email,
                'role' => 'participant',
            ]);
            $added++;
        }
    }

    return response()->json([
        'message' => "{$added} members imported",
        'attendees' => $meeting->attendees()->with('user')->get(),
    ]);
}
```

```js
// resources/js/browser-recorder.js
export default function browserRecorder() {
    return {
        recording: false,
        mediaRecorder: null,
        chunks: [],
        audioUrl: null,
        duration: 0,
        timer: null,

        async startRecording() {
            const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
            this.mediaRecorder = new MediaRecorder(stream);
            this.chunks = [];

            this.mediaRecorder.ondataavailable = (e) => this.chunks.push(e.data);
            this.mediaRecorder.onstop = () => this.processRecording();

            this.mediaRecorder.start();
            this.recording = true;
            this.duration = 0;
            this.timer = setInterval(() => this.duration++, 1000);
        },

        stopRecording() {
            this.mediaRecorder?.stop();
            this.mediaRecorder?.stream.getTracks().forEach(t => t.stop());
            this.recording = false;
            clearInterval(this.timer);
        },

        processRecording() {
            const blob = new Blob(this.chunks, { type: 'audio/webm' });
            this.audioUrl = URL.createObjectURL(blob);
            // Upload via FormData to transcription endpoint
        },

        async uploadRecording(meetingId) {
            const blob = new Blob(this.chunks, { type: 'audio/webm' });
            const formData = new FormData();
            formData.append('audio_file', blob, 'recording.webm');

            const response = await fetch(`/meetings/${meetingId}/transcriptions`, {
                method: 'POST',
                headers: { 'X-CSRF-TOKEN': document.querySelector('meta[name=csrf-token]').content },
                body: formData,
            });

            return response.json();
        },

        formatDuration(seconds) {
            const m = Math.floor(seconds / 60).toString().padStart(2, '0');
            const s = (seconds % 60).toString().padStart(2, '0');
            return `${m}:${s}`;
        },
    };
}
```
